\begin{answer}
    Trivially,
    \[
        \begin{aligned}
\mathbb E_{s\sim p(s)\atop a\sim\pi_0(s, a)} \frac{\pi_1(s, a)}{\pi_0(s, a)} R(s, a) &= \sum_{s, a}p(s)\pi_0(s, a)\frac{\pi_1(s, a)}{\pi_0(s, a)} R(s, a)\\
&=\sum_{s, a}p(s)\pi_1(s, a)R(s, a)\\
            &= \mathbb E_{s\sim p(s)\atop a\sim\pi_0(s, a)}R(s, a)
\end{aligned}
    \]

\end{answer}
